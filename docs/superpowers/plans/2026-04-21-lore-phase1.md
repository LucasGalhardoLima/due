# Lore Phase 1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build Lore Phase 1 — a privacy-first personal link vault for iOS and macOS with on-device AI enrichment, bento grid Chronicle view, and an MCP server for Claude access from day one.

**Architecture:** SwiftData with an App Groups shared container (`group.com.lore.app`) lets the Share Extension write stubs that the main app enriches. A `Scribe` service (LPMetadataProvider + WKWebView + bundled Readability.js + NLContextualEmbedding) runs via `BGProcessingTask`. A `Keeper` service (Foundation Models) generates summaries and PARA Volume classifications. The macOS app writes `lore.sqlite` on every save; the Node.js MCP server reads it — no live connection required.

**Tech Stack:** Swift 6.0+, SwiftUI, SwiftData, Foundation Models (iOS 26+/macOS 15+), NLContextualEmbedding, LPMetadataProvider, WKWebView, BGProcessingTask, XcodeGen, Node.js + better-sqlite3

    case discovered  // URL + title only, enrichment pending
    case canon       // fully enriched

    case projects, areas, resources, archive

    public var displayName: String {
        rawValue.prefix(1).uppercased() + rawValue.dropFirst()
    }

public enum LoreSource: String, Codable, Sendable {
    case safari, threads, linkedin, reddit, other

    public static func detect(from urlString: String) -> LoreSource {
        let s = urlString.lowercased()
        if s.contains("threads.net") { return .threads }
        if s.contains("linkedin.com") { return .linkedin }
        if s.contains("reddit.com") { return .reddit }
        return .safari
    }
}

    public var volume: Volume? {
        get { volumeRaw.flatMap { Volume(rawValue: $0) } }
        set { volumeRaw = newValue?.rawValue }

        self.id = UUID()
        self.url = url
        self.title = title
        self.source = source.rawValue
        self.savedAt = Date()
        self.statusRaw = CanonStatus.discovered.rawValue
        self.tags = []
    }

import Foundation
import SwiftData

public enum CodexStore {
    /// Production container — uses App Groups shared path so the Share Extension
    /// and the main app read from the same store.
    public static func makeContainer() throws -> ModelContainer {
        let schema = Schema([LoreEntry.self])
        if let groupURL = AppConfig.sharedContainerURL {
            let storeURL = groupURL.appending(path: "lore.store")
            let config = ModelConfiguration(schema: schema, url: storeURL)
            return try ModelContainer(for: schema, configurations: [config])
        }
        // Simulator / devices without App Groups provisioned
        return try ModelContainer(for: schema)
    }

    /// In-memory container for unit tests — no disk I/O.
    public static func makeInMemoryContainer() throws -> ModelContainer {
        let schema = Schema([LoreEntry.self])
        let config = ModelConfiguration(schema: schema, isStoredInMemoryOnly: true)
        return try ModelContainer(for: schema, configurations: [config])
    }
}

// Packages/LoreAI/Sources/LoreAI/Keeper.swift
import Foundation
import LoreCore

#if canImport(FoundationModels)
import FoundationModels
#endif

/// Keeper wraps Foundation Models to generate a 1-line summary and assign
/// a PARA Volume for a given piece of content. Degrades gracefully on
/// devices where Foundation Models is unavailable — returns (nil, nil).
public actor Keeper {
    public static let shared = Keeper()
    private init() {}

    public func enrich(title: String, content: String) async -> (summary: String?, volume: Volume?) {
        #if canImport(FoundationModels)
        guard SystemLanguageModel.default.availability == .available else { return (nil, nil) }
        let session = LanguageModelSession()
        async let summary = generateSummary(session: session, title: title, content: content)
        async let volume = classifyVolume(session: session, title: title, content: content)
        return await (summary, volume)
        #else
        return (nil, nil)
        #endif
    }

    #if canImport(FoundationModels)
    private func generateSummary(
        session: LanguageModelSession,
        title: String,
        content: String
    ) async -> String? {
        let excerpt = String(content.prefix(800))
        let prompt = """
        Write one sentence (max 15 words) summarizing this content. No quotes.
        Title: \(title)
        Content: \(excerpt)
        Summary:
        """
        return try? await session.respond(to: prompt).content
            .trimmingCharacters(in: .whitespacesAndNewlines)
    }

    private func classifyVolume(
        session: LanguageModelSession,
        title: String,
        content: String
    ) async -> Volume? {
        let excerpt = String(content.prefix(400))
        let prompt = """
        Reply with exactly one word from: projects areas resources archive
        projects = active work with a deadline
        areas = ongoing responsibilities (health, finances, career, hobbies)
        resources = reference material to revisit
        archive = completed or no-longer-relevant
        Title: \(title)
        Content: \(excerpt)
        Category:
        """
        let raw = try? await session.respond(to: prompt).content
            .trimmingCharacters(in: .whitespacesAndNewlines)
            .lowercased()
        return raw.flatMap { Volume(rawValue: $0) }
    }
    #endif
}
```

// Packages/LoreAI/Sources/LoreAI/VectorIndex.swift
import Foundation
import NaturalLanguage

/// Generates mean-pooled document embeddings using NLContextualEmbedding
/// and provides cosine similarity for semantic search.
public actor VectorIndex {
    public static let shared = VectorIndex()
    private var model: NLContextualEmbedding?
    private init() {}

    private func loadIfNeeded() async {
        guard model == nil else { return }
        let m = NLContextualEmbedding(language: .english)
        try? await m.load()
        model = m
    }

    /// Returns a mean-pooled Float vector for the given text, or nil if
    /// the NLContextualEmbedding model is unavailable on this device.
    public func embed(_ text: String) async -> [Float]? {
        await loadIfNeeded()
        guard let m = model,
              let result = try? m.embeddingResult(for: text, language: .english)
        else { return nil }

        var sum = [Float](repeating: 0, count: m.dimension)
        var count = 0
        result.enumerateTokenVectors(in: text.startIndex..<text.endIndex) { vector, _ in
            for i in 0..<min(vector.count, sum.count) {
                sum[i] += Float(vector[i])
            }
            count += 1
            return true
        }
        guard count > 0 else { return nil }
        let n = Float(count)
        return sum.map { $0 / n }
    }

    public static func encode(_ vector: [Float]) -> Data {
        vector.withUnsafeBufferPointer { Data(buffer: $0) }
    }

    public static func decode(_ data: Data) -> [Float] {
        data.withUnsafeBytes { Array($0.bindMemory(to: Float.self)) }
    }

    public static func cosineSimilarity(_ a: [Float], _ b: [Float]) -> Float {
        guard a.count == b.count else { return 0 }
        let dot = zip(a, b).reduce(Float(0)) { $0 + $1.0 * $1.1 }
        let magA = sqrt(a.reduce(Float(0)) { $0 + $1 * $1 })
        let magB = sqrt(b.reduce(Float(0)) { $0 + $1 * $1 })
        guard magA > 0, magB > 0 else { return 0 }
        return dot / (magA * magB)
    }
}
```

// LoreShareExtension/ShareViewController.swift
import UIKit
import SwiftUI
import SwiftData
import LoreCore

/// Receives a URL from any iOS app, writes a .discovered stub to the
/// shared Codex container, shows "Added to your Codex" for ~1 second,
/// then dismisses. Never performs enrichment — that's the main app's job.
final class ShareViewController: UIViewController {
    override func viewDidLoad() {
        super.viewDidLoad()
        view.backgroundColor = .clear
        Task { await handleShare() }
    }

    private func handleShare() async {
        guard
            let item = extensionContext?.inputItems.first as? NSExtensionItem,
            let provider = item.attachments?.first
        else { return done() }

        // Extract URL
        let urlItem = try? await provider.loadItem(forTypeIdentifier: "public.url")
        guard let url = urlItem as? URL else { return done() }

        // Extract title from selected text or page title hint
        let textItem = try? await provider.loadItem(forTypeIdentifier: "public.plain-text")

// Lore/Services/ReadabilityExtractor.swift
import Foundation
import WebKit

/// Loads a URL in a hidden WKWebView, injects the bundled Readability.js,
/// and returns the extracted article text. Must run on the main actor
/// because WKWebView requires the main thread.
@MainActor
final class ReadabilityExtractor: NSObject, WKNavigationDelegate {
    private var continuation: CheckedContinuation<String?, Never>?
    private var webView: WKWebView?

    private static let script: String = {
        guard
            let url = Bundle.main.url(forResource: "readability", withExtension: "js"),
            let src = try? String(contentsOf: url, encoding: .utf8)
        else { return "" }
        return src
    }()

    /// Extracts readable text from the given URL.
    /// Returns nil on timeout (20s), navigation failure, or unreadable pages.
    func extract(from url: URL) async -> String? {
        await withCheckedContinuation { cont in
            self.continuation = cont
            let wv = WKWebView(frame: .zero)
            wv.navigationDelegate = self
            self.webView = wv
            wv.load(URLRequest(url: url, timeoutInterval: 15))

            Task { @MainActor in
                try? await Task.sleep(for: .seconds(20))
                guard self.continuation != nil else { return }
                self.finish(nil)
            }
        }
    }

    func webView(_ webView: WKWebView, didFinish _: WKNavigation!) {
        let js = """
        (function(){
            \(Self.script)
            var doc = document.cloneNode(true);
            var article = new Readability(doc).parse();
            return article ? article.textContent.trim() : null;
        })();
        """
        webView.evaluateJavaScript(js) { result, _ in
            self.finish(result as? String)
        }
    }

    func webView(_ webView: WKWebView, didFail _: WKNavigation!, withError _: Error) {
        finish(nil)
    }

    func webView(_ webView: WKWebView,
                 didFailProvisionalNavigation _: WKNavigation!,
                 withError _: Error) {
        finish(nil)
    }

    private func finish(_ text: String?) {
        continuation?.resume(returning: text)
        continuation = nil
        webView = nil
    }
}

// Lore/Services/Scribe.swift
import Foundation
import SwiftData
import LinkPresentation
import LoreCore
import LoreAI

/// Processes all .discovered entries in the Codex:
/// LPMetadataProvider → WKWebView/Readability → NLEmbedding → Keeper → .canon
@MainActor
final class Scribe {
    static let shared = Scribe()
    private init() {}

    func processDiscoveredEntries(context: ModelContext) async {
        let descriptor = FetchDescriptor<LoreEntry>(
            predicate: #Predicate { $0.statusRaw == "discovered" }
        )
        let entries = (try? context.fetch(descriptor)) ?? []
        for entry in entries {
            await enrich(entry: entry, context: context)
        }
    }

    private func enrich(entry: LoreEntry, context: ModelContext) async {
        guard let url = URL(string: entry.url) else {
            entry.status = .canon  // malformed URL — don't retry forever
            try? context.save()
            return
        }

        // 1. LPMetadataProvider — hero image
        if let imageURL = await fetchHeroImageURL(url: url) {
            entry.heroImagePath = await cacheImage(from: imageURL, id: entry.id)
        }

        // 2. WKWebView + Readability.js — full article text
        let extractor = ReadabilityExtractor()
        entry.content = await extractor.extract(from: url)

        // 3. NLContextualEmbedding — semantic vector
        let textForEmbedding = [entry.title, entry.content]
            .compactMap { $0 }.joined(separator: ". ")
        if let vector = await VectorIndex.shared.embed(textForEmbedding) {
            entry.embeddingData = VectorIndex.encode(vector)
        }

        // 4. Keeper (Foundation Models) — summary + Volume
        let contentForAI = entry.content ?? entry.title
        let (summary, volume) = await Keeper.shared.enrich(
            title: entry.title,
            content: contentForAI
        )
        entry.aiSummary = summary
        entry.volume = volume

        entry.status = .canon
        try? context.save()

        // Trigger SQLite export on macOS after each enrichment
        #if os(macOS)
        await SQLiteExporter.shared.export(context: context)
        #endif
    }

    // MARK: Helpers

    private func fetchHeroImageURL(url: URL) async -> URL? {
        let provider = LPMetadataProvider()
        guard let metadata = try? await provider.startFetchingMetadata(for: url),
              let imageProvider = metadata.imageProvider
        else { return nil }
        return try? await imageProvider.loadObject(ofClass: URL.self)
    }

    private func cacheImage(from imageURL: URL, id: UUID) async -> String? {
        guard let data = try? Data(contentsOf: imageURL) else { return nil }
        let dir = FileManager.default.urls(for: .cachesDirectory, in: .userDomainMask)[0]
        let file = dir.appending(path: "lore_\(id.uuidString).jpg")
        try? data.write(to: file)
        return file.path
    }
}

// Lore/Services/SQLiteExporter.swift
#if os(macOS)
import Foundation
import SQLite3
import SwiftData
import LoreCore

/// Writes a denormalized lore.sqlite to ~/Library/Application Support/Lore/
/// after every vault change. The MCP server reads this file.
@MainActor
final class SQLiteExporter {
    static let shared = SQLiteExporter()
    private init() {}

    private var exportURL: URL? {
        FileManager.default
            .urls(for: .applicationSupportDirectory, in: .userDomainMask)
            .first?
            .appending(path: "Lore/lore.sqlite")
    }

    func export(context: ModelContext) async {
        guard let dest = exportURL else { return }
        try? FileManager.default.createDirectory(
            at: dest.deletingLastPathComponent(),
            withIntermediateDirectories: true
        )
        let entries = (try? context.fetch(FetchDescriptor<LoreEntry>())) ?? []
        writeDB(to: dest.path, entries: entries)
    }

    private func writeDB(to path: String, entries: [LoreEntry]) {
        var db: OpaquePointer?
        guard sqlite3_open(path, &db) == SQLITE_OK else { return }
        defer { sqlite3_close(db) }

        sqlite3_exec(db, """
            CREATE TABLE IF NOT EXISTS lore_entries (
                id TEXT PRIMARY KEY,
                url TEXT NOT NULL,
                title TEXT NOT NULL,
                content TEXT,
                ai_summary TEXT,
                source TEXT,
                saved_at TEXT,
                status TEXT,
                volume TEXT,
                tags TEXT,
                embedding BLOB
            );
        """, nil, nil, nil)

        sqlite3_exec(db, "BEGIN; DELETE FROM lore_entries;", nil, nil, nil)

        let sql = """
        INSERT INTO lore_entries
        (id,url,title,content,ai_summary,source,saved_at,status,volume,tags,embedding)
        VALUES (?,?,?,?,?,?,?,?,?,?,?);
        """
        var stmt: OpaquePointer?
        sqlite3_prepare_v2(db, sql, -1, &stmt, nil)
        let iso = ISO8601DateFormatter()

        for e in entries {
            bind(stmt, 1, e.id.uuidString)
            bind(stmt, 2, e.url)
            bind(stmt, 3, e.title)
            bind(stmt, 4, e.content)
            bind(stmt, 5, e.aiSummary)
            bind(stmt, 6, e.source)
            bind(stmt, 7, iso.string(from: e.savedAt))
            bind(stmt, 8, e.statusRaw)
            bind(stmt, 9, e.volumeRaw)
            bind(stmt, 10, e.tags.isEmpty ? nil : e.tags.joined(separator: ","))
            if let data = e.embeddingData {
                sqlite3_bind_blob(stmt, 11, (data as NSData).bytes, Int32(data.count), nil)
            } else {
                sqlite3_bind_null(stmt, 11)
            }
            sqlite3_step(stmt)
            sqlite3_reset(stmt)
        }

        sqlite3_finalize(stmt)
        sqlite3_exec(db, "COMMIT;", nil, nil, nil)
    }

    private func bind(_ stmt: OpaquePointer?, _ idx: Int32, _ value: String?) {
        if let v = value {
            sqlite3_bind_text(stmt, idx, (v as NSString).utf8String, -1, nil)
        } else {
            sqlite3_bind_null(stmt, idx)
        }
    }
}
#endif

@Observable
@MainActor
final class ChronicleViewModel {
    var searchText = ""
    private(set) var searchResults: [LoreEntry] = []

    func search(query: String, context: ModelContext) async {
        let q = query.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !q.isEmpty else { searchResults = []; return }

        // Semantic first
        if let queryVec = await VectorIndex.shared.embed(q) {
            let all = (try? context.fetch(FetchDescriptor<LoreEntry>())) ?? []
            let scored: [(LoreEntry, Float)] = all.compactMap {
                guard let data = $0.embeddingData else { return nil }
                let score = VectorIndex.cosineSimilarity(queryVec, VectorIndex.decode(data))
                return score > 0.25 ? ($0, score) : nil
            }
            if !scored.isEmpty {
                searchResults = scored.sorted { $0.1 > $1.1 }.map(\.0)
                return
            }
        }

        // Keyword fallback
        let lower = q.lowercased()
        let desc = FetchDescriptor<LoreEntry>(predicate: #Predicate {
            $0.title.localizedStandardContains(lower) ||
            ($0.aiSummary ?? "").localizedStandardContains(lower)
        })
        searchResults = (try? context.fetch(desc)) ?? []
    }
}

```javascript
// mcp-server/src/index.js
import { Server } from "@modelcontextprotocol/sdk/server/index.js";
import { StdioServerTransport } from "@modelcontextprotocol/sdk/server/stdio.js";
import { CallToolRequestSchema, ListToolsRequestSchema } from "@modelcontextprotocol/sdk/types.js";
import Database from "better-sqlite3";
import { homedir } from "os";
import { join } from "path";
import { existsSync } from "fs";

const DB_PATH = join(homedir(), "Library", "Application Support", "Lore", "lore.sqlite");

function db() {
  if (!existsSync(DB_PATH)) {
    throw new Error(`Lore database not found at ${DB_PATH}. Open the Lore macOS app to generate it.`);
  }
  return new Database(DB_PATH, { readonly: true });
}

const server = new Server(
  { name: "lore-mcp-server", version: "1.0.0" },
  { capabilities: { tools: {} } }
);

server.setRequestHandler(ListToolsRequestSchema, async () => ({
  tools: [
    {
      name: "search",
      description: "Keyword search across all Lore Codex entries (title, summary, content).",
      inputSchema: {
        type: "object",
        properties: {
          query: { type: "string" },
          limit: { type: "number", default: 10 }
        },
        required: ["query"]
      }
    },
    {
      name: "get_entry",
      description: "Get full content of a Lore entry by URL or UUID.",
      inputSchema: {
        type: "object",
        properties: { identifier: { type: "string" } },
        required: ["identifier"]
      }
    },
    {
      name: "list_volume",
      description: "List entries in a PARA Volume: projects, areas, resources, or archive.",
      inputSchema: {
        type: "object",
        properties: {
          volume: { type: "string", enum: ["projects", "areas", "resources", "archive"] },
          limit: { type: "number", default: 20 }
        },
        required: ["volume"]
      }
    },
    {
      name: "list_recent",
      description: "List the most recently saved entries.",
      inputSchema: {
        type: "object",
        properties: { limit: { type: "number", default: 10 } }
      }
    }
  ]
}));

server.setRequestHandler(CallToolRequestSchema, async ({ params: { name, arguments: args } }) => {
  const text = (data) => ({ content: [{ type: "text", text: JSON.stringify(data, null, 2) }] });

  if (name === "search") {
    const q = `%${args.query.toLowerCase()}%`;
    const rows = db().prepare(`
      SELECT id, url, title, ai_summary AS summary, volume, saved_at AS savedAt
      FROM lore_entries
      WHERE lower(title) LIKE ? OR lower(ai_summary) LIKE ? OR lower(content) LIKE ?
      ORDER BY saved_at DESC LIMIT ?
    `).all(q, q, q, args.limit ?? 10);
    return text(rows);
  }

  if (name === "get_entry") {
    const row = db().prepare(
      `SELECT * FROM lore_entries WHERE url = ? OR id = ? LIMIT 1`
    ).get(args.identifier, args.identifier);
    if (!row) return text({ error: "Entry not found" });
    return text({ ...row, tags: row.tags ? row.tags.split(",") : [] });
  }

  if (name === "list_volume") {
    const rows = db().prepare(`
      SELECT id, url, title, ai_summary AS summary, saved_at AS savedAt
      FROM lore_entries WHERE volume = ? ORDER BY saved_at DESC LIMIT ?
    `).all(args.volume, args.limit ?? 20);
    return text(rows);
  }

  if (name === "list_recent") {
    const rows = db().prepare(`
      SELECT id, url, title, ai_summary AS summary, volume, saved_at AS savedAt
      FROM lore_entries ORDER BY saved_at DESC LIMIT ?
    `).all(args.limit ?? 10);
    return text(rows);
  }

  throw new Error(`Unknown tool: ${name}`);
});

await server.connect(new StdioServerTransport());
```
